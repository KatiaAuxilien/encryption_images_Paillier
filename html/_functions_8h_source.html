46// Génère un kernel de flou gaussien selon la taille et le sigma passés en paramètres
47void generateGaussianKernel(std::vector<std::vector<float>> & kernel, int size, float sigma) {
48 float sum = 0.0;
49 for (int y = -size/2; y <= size/2; ++y) {
50 for (int x = -size/2; x <= size/2; ++x) {
51 float value = exp(-(x*x + y*y) / (2 * sigma*sigma)) / (2 * M_PI * sigma*sigma);
52 kernel[y + size/2][x + size/2] = value;
53 sum += value;
54 }
55 }
56 for (int y = 0; y < size; ++y) {
57 for (int x = 0; x < size; ++x) {
58 kernel[y][x] /= sum;
59 }
60 }
61}

106// droite (dir == 0) ou de droite à gauche (dir == 1) et stocke les images respectives dans o_images
107void selectiveProgressiveEncryption(ImageBase & image, ImageBase o_images[8], bool MSBtoLSB) {
108 unsigned int bSeq[8];
109 int height, width;
110 height = image.getHeight();
111 width = image.getWidth();
112 for (int i = 0; i < height; ++i) {
113 for (int j = 0; j < width; ++j) {
114 genPRNS(bSeq);
115 std::bitset<8> pBits_R(image[3*i][3*j+0]);
116 std::bitset<8> pBits_G(image[3*i][3*j+1]);
117 std::bitset<8> pBits_B(image[3*i][3*j+2]);
118 for (int b = 0; b < 8; ++b) {
119 if (MSBtoLSB) {
120 pBits_R[7 - b] = (pBits_R[7 - b] ^ bSeq[b]);
121 pBits_G[7 - b] = (pBits_G[7 - b] ^ bSeq[b]);
122 pBits_B[7 - b] = (pBits_B[7 - b] ^ bSeq[b]);
123 }
124 else {
125 pBits_R[b] = (pBits_R[b] ^ bSeq[7 - b]);
126 pBits_G[b] = (pBits_G[b] ^ bSeq[7 - b]);
127 pBits_B[b] = (pBits_B[b] ^ bSeq[7 - b]);
128 }
129 o_images[b][3*i][3*j+0] = pBits_R.to_ulong();
130 o_images[b][3*i][3*j+1] = pBits_G.to_ulong();
131 o_images[b][3*i][3*j+2] = pBits_B.to_ulong();

188void areaScrambling(ImageBase & image, ImageBase & o_image, int start_i, int start_j, int area_h, int area_w) {
189 int nbPixels = area_h * area_w;
190 std::random_device randev;
191 std::mt19937 rng(randev());
192 std::uniform_int_distribution<unsigned int> distrbit(0, nbPixels-1);
193 ImageBase unavailability(area_w, area_h, false);
194 for (int i = 0; i < area_h; ++i)
195 for (int j = 0; j < area_w; ++j) {
196 int randIdx = distrbit(rng);
197 int randJ = randIdx % area_w;
198 int randI = (randIdx - randJ) / area_w;
199 while (unavailability[randI][randJ] == 1) {
200 randIdx = (randIdx == nbPixels-1 ? 0 : randIdx+1);
201 randJ = randIdx % area_w;
202 randI = (randIdx - randJ) / area_w;
203 }
204 unavailability[randI][randJ] = 1;
205 o_image[3 * (start_i + randI)][3 * (start_j + randJ) + 0] = image[3 * (start_i + i)][3 * (start_j + j) + 0];
206 o_image[3 * (start_i + randI)][3 * (start_j + randJ) + 1] = image[3 * (start_i + i)][3 * (start_j + j) + 1];
207 o_image[3 * (start_i + randI)][3 * (start_j + randJ) + 2] = image[3 * (start_i + i)][3 * (start_j + j) + 2];
208 }

336double computePSNR(ImageBase & image1, ImageBase & image2) {
337 double mse = 0;
338 double maxIntensity = 255.0;
339
340 if (image1.getHeight() != image2.getHeight() || image1.getWidth() != image2.getWidth()) {
341 printf("Les images ne sont pas de la même taille !\n");
342 return -1;
343 }
344
345 for (int i = 0; i < image1.getHeight(); ++i)
346 for (int j = 0; j < image1.getWidth(); ++j)
347 for (int c = 0; c < 3; ++c)
348 mse += pow((image1[3*i][3*j+c] - image2[3*i][3*j+c]), 2);
349
350 mse /= (image1.getHeight() * image1.getWidth() * 3);
351 double psnr = 10 * log10(pow(maxIntensity, 2) / mse);
352 return psnr;
353}
