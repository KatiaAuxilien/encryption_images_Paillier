72 uint64_t fastMod_64t(uint64_t x, uint64_t e, uint64_t n)
73 {
74 uint64_t c = 1;
75 bitset<BITSETSIZE> bits = bitset<BITSETSIZE>(e);
76
77 for (int i = BITSETSIZE - 1; i >= 0; i--)
78 {
79 c = c * c % n;
